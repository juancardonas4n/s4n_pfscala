\subsection{Definición de funciones puras}
\label{sec:def-fun-puras}

\begin{frame}
  \frametitle{Módulo 2. Funciones puras}
  \framesubtitle{Definición de funciones puras}

  \begin{quote}
    Una \hl{función pura} es una función que depende únicamente de sus parámetros de entrada y su algoritmo produce
    su salida. Este no lee cualquier valor del ``mundo exterior'' y no modifica valor alguno del mundo exterior.
  \end{quote}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Módulo 2. Funciones puras}
  \framesubtitle{Definición de funciones}
   \begin{itemize}
   \item<1-> Scala nos permite definir valores inmutables \texttt{val}
% \begin{lstlisting}[language=Haskell]
% f (x,y) = (x, ['a' .. y])
% \end{lstlisting}
\begin{lstlisting}[language=Scala]
val diez = 10
\end{lstlisting}
\item<2-> Scala nos permite definir funciones. Definamos una
  función que calcule el mínimo de dos valores enteros.
\begin{lstlisting}[language=Scala]
def minimo(x:Int, y:Int):Int = if (x<=y) x else y
\end{lstlisting}
   \end{itemize}
 \end{frame}


 \begin{frame}
  \frametitle{Módulo 2. Funciones puras}
  \framesubtitle{Definición de funciones}

  \begin{exampleblock}{Definición de signo}
    Defina una función \texttt{signo} que evalue la siguiente definición:
    \[
      signo(x) = \begin{cases}
        x < 0 & -1 \\
        x = 0 & 0 \\
        x > 0 & 1 \\
        \end{cases}
    \]
  \end{exampleblock}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Módulo 2. Funciones puras}

  \begin{exampleblock}{Solucion \texttt{signo}}
\begin{lstlisting}[language=scala]
def signo(x:Int):Int = if (x == 0) 0 
                       else if (x > 0) 1 else -1
\end{lstlisting}
  \end{exampleblock}
  
\end{frame}

\subsection{Números fraccionarios}

\begin{frame}
  \frametitle{Módulo 2. Funciones puras}
  \framesubtitle{Números fraccionarios}

  \begin{block}{Definición}
    Según Wikipedia:
    \begin{quote}
      En matemáticas, una fracción, número fraccionario, $\ldots$ es
      la expresión de una cantidad dividida entre otra cantidad; es
      decir que representa un cociente no efectuado de números. Por
      razones históricas también se les llama fracción común, fracción
      mixta o fracción decimal. Las fracciones comunes se componen de:
      numerador, denominador y línea divisora entre ambos (barra
      horizontal u oblicua). En una fracción común $a/b$ el
      denominador "b" expresa la cantidad de partes iguales que
      representan la unidad, y el numerador "a" indica cuántas de
      ellas se toman.
    \end{quote}
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{Módulo 2. Funciones puras}
  \framesubtitle{Números fraccionarios}

  \begin{exampleblock}{Ejercicio}
    Defina una función llamada de identidad \texttt{idFraccion} que
    reciba un número fraccionario y devuelva el mismo número
    fraccionario.
  \end{exampleblock}

  \begin{exampleblock}{Ejercicio}
    Defina una función llamada de identidad \texttt{idInfoEstudiante}
    que reciba un estudiante y devuelva el mismo estudiante.
  \end{exampleblock}
\end{frame}

\begin{frame}
  \frametitle{Módulo 2. Funciones puras}
  \framesubtitle{Números fraccionarios}

  \begin{block}<1->{Definición fracción propia}
    Según Wikipedia:
    \begin{quote}
      Las fracciones comunes pueden clasificarse en propias e impropias. Una \textbf{fracción propia} es
      aquella en la que, si el numerador y el denominador son positivos, el numerador es menor que el denominador, por
      ejemplo: $\frac{1}{3}$, $\frac{3}{8}$, $\frac{3}{4}$.
      El contrario define una \texttt{fracción impropia}.
    \end{quote}
    \begin{exampleblock}<2->{Ejercicio} Defina una función \texttt{esPropia}
      para establecer si una fracción es propia (\texttt{true}), en
      caso contrario (\texttt{false}).
    \end{exampleblock}
  \end{block}
\end{frame}
